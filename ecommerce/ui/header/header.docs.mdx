import { Header } from './header';
import { MemoryRouter } from 'react-router';

A header component that renders everything needed for the header including logo and navigation links

### Simple Header

```js live
<MemoryRouter>
  <Header
    logoText="Bit Sports Store"
    src="https://static.bit.dev/bit-logo.svg"
    alt="Bit Logo"
  ></Header>
</MemoryRouter>
```

### Header with Navigation

```js live
<MemoryRouter>
  <Header
    logoText="Bit Sports Store"
    src="https://static.bit.dev/bit-logo.svg"
    alt="Bit Logo"
  >
    add your links here
  </Header>
</MemoryRouter>
```
