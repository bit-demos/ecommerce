import { Footer } from './footer';
import { MemoryRouter } from 'react-router';

A footer component that renders everything needed for the footer including logo and navigation links

### Simple Footer

```js live
<MemoryRouter>
  <Footer
    logoText="Bit Perfume Store"
    src="https://static.bit.dev/bit-logo.svg"
    alt="Bit Logo"
  ></Footer>
</MemoryRouter>
```

### Footer wih Nav Links

```js live
<MemoryRouter>
  <Footer
    logoText="Bit Perfume Store"
    src="https://static.bit.dev/bit-logo.svg"
    alt="Bit Logo"
  >
    Links go here
  </Footer>
</MemoryRouter>
```
